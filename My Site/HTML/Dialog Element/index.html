<script>
    var modal = document.querySelector("dialog");
    modal.show();                                                                      // opens the dialog box (same as the "open" attribute)
    modal.showModal();                                                                 // opens the dialog box (as modal window "full screen-mode" | creates a backdrop element | user can close the modal by pressing "Esc" button )  
    
    modal.close("box closed");                                                         // closes the dialog box (message is returned (optionnal))
    modal.returnValue                                                                  // -> "box closed" (returns the value passed by the "close()" method)
</script>

-- CSS --
    .myModal::backdrop {                                                               // style only the backdrop (dialog box opened with the "showModal()" method)
        background-color:red;
    }

    <dialog id="myDialogBox">myDialog Box</dialog>
    <p><button onmouseup="openIt(event)"> "open()" </button> (opens the dialog box (same as the <mark>open</mark> attribute) <u>[user can interact with the rest of the page]</u>)</p>
    <p><button onmouseup="openModal(event)"> "openModal()" </button> (opens the dialog box <u>with backdrop [user CANNOT interact with the rest of the page | <span style="background-color:yellow;">box can be closed by pressing the "Esc" button on the keyboard </span>]</u> </p>   
    <p><button onmouseup="closeIt(event)"> "close()" </button> (closes the dialog box (oppened by the <mark>show()</mark> method or by the <mark>open</mark> attribute))</p>
    <p class="mBox_status"><u> Dialog Box Status : <span style="color:yellow; font-weight:bold;"></span> </u></p>
<script>
    var dBox = document.querySelector("#myDialogBox");

    function openIt(event) {
        dBox.show();
        // dBox.open = true;                <- this does the same
    }
    function openModal(event) {
        dBox.showModal();
    }
    function closeIt(event) {
        dBox.close();
        // dBox.open = false;               <-- this does the same
        document.getElementsByClassName("mBox_message")[0].firstElementChild.firstElementChild.textContent = dBox.returnValue;
    }
</script>
