obj.__dict__ - returns a dictionary which holds the object's own attributes (called mappingproxy)
- gives full access to the object's attributes (we can get|set object attributes through it)
getattr(obj, attr:str) - returns the object's attribute value
hasattr(obj, attr:str) - returns True if the object has the specified attribute, return False otherwise
setattr(obj, attr:str, val:any) - sets the specified attribute to the object
delattr(obj, attr:str) - deletes the specified attribute from the object
del obj.attr - deletes the object's attribute

DEMO 
    class Object(object):
        pass
    
    obj = Object()
    
    obj.attr = 21                                                                      // setting attribute value 
    obj.method = lambda x : x**2
    
    obj.attr                                                                           // -> 21         // get attribute value 
    obj.method(3)                                                                      // -> 9
    
    del obj.method                                                                     // delete object attribute   
    hasattr(obj, 'method')                                                             // -> False 
    
    
// instances derived directly from the object class cannot have attributes ------------
    obj = object()                                                                     // instance cannot have any attributes   
    
    obj.attr = 21                                                                      // -! raises an AttributeError 
    
    
// objects are mutable ----------------------------------------------------------------
    class Object(object):
        pass
        
    obj1 = Object()
    obj1.attr = 21
    
    obj2 = obj1                                                                        // obj2 and obj1 varialbes point to the same underlying data (objects are mutalbe)
    
    obj2.attr = 22
    obj1.__dict__attr                                                                  // -> {'attr': 22}
    obj2.__dict__attr                                                                  // -> {'attr': 22}
    
  // primitives are immutable -----------------
    prm1 = 21
    prm2 = prm1                                                                        // prm2 creates its own copy from prm1 (primitives are immutable)  
    
    prm2 = 22
    prm1                                                                               // -> 21
    prm2                                                                               // -> 22

- __dict__ 

    class Object(object):
        pass
        
    obj1 = Object()
    obj1.attr = 21 
    
    obj1.__dict__                                                                      // -> {'attr': 21}         // basically gives full access to the object's attributes as dictionary   
    
    obj1.__dict__["attr2"] = 22                                                        // setts a new object attribute through the mapping proxy 
    obj1.__dict__                                                                      // -> {'attr': 21, 'attr2': 22}
    
    del obj1.__dict__["attr"]                                                          // deleting object attributes   
    del obj1.__dict__["attr2"]
    
    obj1.__dict__                                                                      // -> {}

- getattr() / hasattr() / setattr() / delattr() / del 

    class Object(object):
        pass
        
    obj1 = Object()

    setattr(obj1, 'attr1', 21)                                                         // sets new attributes 
    setattr(obj1, 'attr2', 22)
    
    getattr(obj1, 'attr1')                                                             // -> 21                   // returns the object attribute's value    
    
    hasattr(obj1, 'attr2')                                                             // -> True                 // check if the object has the specified attribute    
    hasattr(obj1, 'attr3')                                                             // -> False
    
    delattr(obj1, 'attr1')                                                             // deletes the specified attribute   
    del obj1.attr2 
    
    obj1.__dict__                                                                      // -> {}
