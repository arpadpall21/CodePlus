- datetime.timedelta() 

    import datetime
    
    timeNow = datetime.datetime.now()
    
    timedelta = datetime.timedelta(
        days=13,
        weeks=1,
        hours=5,
        minutes=5
    )
    
    print( timedelta )                // -> returns a time object (passed time)
    print( timeNow + timedelta )      // -> the returned object is the current time added to the passed time (timedelta)
