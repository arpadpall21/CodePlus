sorted(iterable, key=fn

Default: None

- function takes 1 argument and must return a number

- the returned number is bound to the checked item, then the sort order is defined according to the bound numbers order

    someList = ['xxxx', 'x', 'xx']

                                // the sort order is defined by the returned numbers -> 1 | 2 | 4
                                // -> ['x', 'xx', 'xxxx']

The sorted() method

DEMO 
// sorted() ---------------------------------------------------------------------------
    myList1 = [111, 1, 3, 15, 1.1]
    sorted(myList1)                                                                    // -> [1, 1.1, 3, 15, 111]    
    myList1                                                                            // -> [111, 1, 3, 15, 1.1]                 // original not modified 
    
    myTuple = ['kcp', 'AB', 'ab', 'kac', 'KCP']
    sorted(myTuple)                                                                    // -> ['AB', 'KCP', 'ab', 'kac', 'kcp']    // sort strings in ASCII transition order   
    
    myString = 'developer'
    sorted(myString)                                                                   // -> ['d', 'e', 'e', 'e', 'l', 'o', 'p', 'r', 'v']    
    
    myList2 = ['kcp', 111, 1, 'AB', 3, 15, 'ab', 'kac', 'KCP']
    sorted(myList2)                                                                    // -! raises a TypeError because the sort does not support sort between strings and numbers    
    
    myList3 = [[11, 22], [3, 4], [1, 2]]
    sorted(myList3)                                                                    // -> [[1, 2], [3, 4], [11, 22]]
    
    myList4 = [{'p33':33}, {"p1":1, "p3":3}]
    sorted(myList4)                                                                    // -! raises a TypeError because the sort does not support sort between dictionaries     
    
    myTuple2 = ('Lorem', 'ipsum', 'dolor', 'sit', 'amet', 'consectetur', 'adipiscing')
    sorted(myTuple2, key=len)                                                          // -> ['sit', 'amet', 'Lorem', 'ipsum', 'dolor', 'adipiscing', 'consectetur']    
                                                                                       // items are sorted by length (the len function returns the length for each string)    
    myTuple3 = (4, 145, 844, 2, 12, 10, -4)
    def sort10(nr):                                                                    // custom sort function, sort numbers lower than 10 left, larger than 10 right   
        if nr < 10:
            return -1
        elif nr > 10:
            return 1
        else:
            return 0
    sorted(myTuple3, key=sort10)                                                       // -> [4, 2, -4, 10, 145, 844, 12]    
    
    myTuplew = (4, 145, 844, 2, 12, 10, -4)
    sorted(myTuplew, reverse=True)                                                     // -> [844, 145, 12, 10, 4, 2, -4]
                                                                                       // sorts the list in descending order

The char() / ord() methods

DEMO 
    'this is a \'test\''                                                               // the ' chacters in the string are escaped (printed as this is a 'test')  
    
// referencing characters from latin-1 table ------------------------------------------
    '\x21'                                                                             // -> '!'        // references the ! character from the latin-1 table as hexadecimal notation    
    '\41'                                                                              // -> '!'        // references the ! character from the latin-1 table as octal notation    
    
    
// referencing characters from UTF-16 unicode table -----------------------------------
    '\u0021'                                                                           // -> '!'        // references the ! character from the UTF-16 unicode table as hexadecimal notation    
    chr(33)                                                                            // -> '!'        // references the ! character from the UTF-16 unicode table as decimal notation    
    ord('!')                                                                           // -> 33         // returns the passed character's decimal notation from the UTF-16 unicode table   
    
    '\u1fa7'                                                                           // -> 'á¾§'
    chr(8103)                                                                          // -> 'á¾§'
    ord('á¾§')                                                                           // -> 8103
    
    
// UTF-16 2 byte character reference limitation ---------------------------------------
    // -! the ðŸ’” character is referenced as d83ddc94 hexadecimal or 3627932820 decimal in the unicode table (encoded in 4 bytes), so the below tools cannot reference it because they are limited to reference only 2 byte characters   
    '/ud83ddc94'                                                                       // -> '/ud83ddc94'
    chr(3627932820)                                                                    // -! raises an OverflowError
